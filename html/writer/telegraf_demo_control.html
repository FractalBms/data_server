Filtered /tmp/telegraf-46-filt.conf
# Filter A + Filter B (starlark float scaling + dedup) — Conservative+

[agent]
  interval       = "1s"
  precision      = "1ms"
  omit_hostname  = true

[[inputs.mqtt_consumer]]
  servers     = ["tcp://localhost:1883"]
  topics      = ["ems/#"]
  client_id   = "telegraf-46-filt"
  data_format = "json_v2"

    # ── Filter A: drop string dtype ──────────────────
    [inputs.mqtt_consumer.tagdrop]
      data_type = ["string"]    ← ~4% reduction
    # ─────────────────────────────────────────────────

  [[inputs.mqtt_consumer.json_v2]]
    timestamp_path   = "ts"
    timestamp_format = "2006-01-02T15:04:05.000Z"

    [[inputs.mqtt_consumer.json_v2.field]]
      path = "value"

    [[inputs.mqtt_consumer.topic_parsing]]
      topic       = "ems/site/+/unit/+/+/+/+/+"
      tags        = "_/_/site_id/_/unit_id/…/data_type"
      measurement = "_/_/_/measurement/…"

    # ── Filter B step 1: float×100 → float_i (order=0) ───
    # Scales floats to int so dedup can exact-match them.
    # On read: divide float_i by 100 to recover value.
  [[processors.starlark]]
      order = 0
      source = '''
    def apply(metric):
        if metric.tags.get("data_type") != "float":
            return metric
        v = metric.fields.get("value")
        if v == None: return metric
        metric.fields["value"] = int(float(v) * 100)
        metric.tags["data_type"] = "float_i"
        return metric
    '''

    # ── Filter B step 2: dedup int/bool/float_i (order=1) ─
  [[processors.dedup]]
      order          = 1
      dedup_interval = "60s"
      [processors.dedup.tagpass]
      data_type = ["integer", "boolean_integer", "float_i"]
    # ─────────────────────────────────────────────────────

[[processors.pivot]]
  order     = 1
  tag_key   = "data_type"
  value_key = "value"

[[outputs.influxdb_v2]]
  urls    = ["http://localhost:8097"]
  bucket  = "ems-filtered"
  namedrop = ["internal_*"]
Filter A — drop string rows at input (~4%)
Filter B — float×100→float_i (Starlark) then dedup int/bool/float_i 60s (~23%+)
