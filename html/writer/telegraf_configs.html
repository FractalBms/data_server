▶ telegraf-conf-v2.conf v2 — baseline · single InfluxDB · concurrent_writes=2 
[agent]
  omit_hostname = true
  precision     = "1ms"
  debug         = {{ .Values.telegraf.debug }}
  flush_interval      = "5s"
  metric_batch_size   = 50000
  metric_buffer_limit = 200000

[[inputs.internal]]
  collect_memstats = false

[[inputs.mqtt_consumer]]
  servers  = ["$MQTT_HOST"]
  topics   = ["$share/telegrafs-v2/ems/#"]
  client_id = "telegraf-v2-${HOSTNAME}"
  qos = 0
  max_undelivered_messages = 1000000
  data_format = "json_v2"

  [[inputs.mqtt_consumer.json_v2]]
    timestamp_path     = "ts"
    timestamp_format   = "2006-01-02T15:04:05Z"
    timestamp_timezone = "UTC"
      # 6/7/8/9-segment topic parsing (same in all configs — omitted for brevity)

[[processors.pivot]]
  order     = 1
  tag_key   = "data_type"
  value_key = "value"

[[outputs.prometheus_client]]
  listen = ":9273"
  path   = "/metrics"

[[outputs.influxdb_v2]]
  urls         = ["$INFLUX_HOST"]
  token        = "$INFLUX_TOKEN"
  organization = "$INFLUX_ORG"
  bucket       = "$INFLUX_BUCKET"
  namedrop         = ["internal_*"]
  content_encoding = "gzip"
  timeout          = "10s"
  concurrent_writes = 2

[agent]
  omit_hostname = true
  precision     = "1ms"
  debug         = {{ .Values.telegrafV3.debug | default false }}  # v3 key
  flush_interval      = "5s"
  metric_batch_size   = 50000
  metric_buffer_limit = 200000

[[inputs.mqtt_consumer]]
  servers   = ["$MQTT_HOST"]
  topics    = ["$share/telegrafs-v3/ems/#"]   # v3 share group
  client_id = "telegraf-v3-${HOSTNAME}"
  qos = 0
  max_undelivered_messages = 1000000
  data_format = "json_v2"
    # topic parsing identical to v2 — omitted for brevity

################################################################################
# PROCESSORS — v3 adds converters BEFORE pivot
################################################################################

[[processors.converter]]  # float cast
  order = 1
  [processors.converter.tagpass]
    data_type = ["float"]
  [processors.converter.fields]
    float = ["value"]

[[processors.converter]]  # integer + boolean_integer cast
  order = 1
  [processors.converter.tagpass]
    data_type = ["integer", "boolean_integer"]
  [processors.converter.fields]
    integer = ["value"]

[[processors.converter]]  # string cast
  order = 1
  [processors.converter.tagpass]
    data_type = ["string"]
  [processors.converter.fields]
    string = ["value"]

[[processors.pivot]]
  order     = 2  # runs AFTER converters
  tag_key   = "data_type"
  value_key = "value"

[[outputs.prometheus_client]]
  listen = ":9273"
  path   = "/metrics"

  token        = "$INFLUX_TOKEN"
  organization = ""  # empty — IOx / serverless InfluxDB (no org concept)
  bucket       = "$INFLUX_BUCKET"
  namedrop         = ["internal_*"]
  content_encoding = "gzip"
  timeout          = "10s"
  concurrent_writes = 2
