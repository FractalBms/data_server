Baseline /tmp/telegraf-46-base.conf
# All signals — no filtering

[agent]
  interval          = "1s"
  precision         = "1ms"
  flush_interval    = "5s"
  omit_hostname     = true

[[inputs.mqtt_consumer]]
  servers    = ["tcp://localhost:1883"]
  topics     = ["ems/#"]
  client_id  = "telegraf-46-base"
  data_format = "json_v2"

    # no tagdrop here — strings pass through

  [[inputs.mqtt_consumer.json_v2]]
    timestamp_path   = "ts"
    timestamp_format = "2006-01-02T15:04:05.000Z"

    [[inputs.mqtt_consumer.json_v2.field]]
      path = "value"

        # 9-seg topic parsing
    [[inputs.mqtt_consumer.topic_parsing]]
      topic = "ems/site/+/unit/+/+/+/+/+"
      tags  = "_/_/site_id/_/unit_id/device/instance/point_name/data_type"
      measurement = "_/_/_/measurement/_/_/_/_/_"

[[processors.pivot]]
  order     = 1
  tag_key   = "data_type"
  value_key = "value"

[[outputs.influxdb_v2]]
  urls         = ["http://localhost:8096"]
  bucket       = "ems-baseline"
  namedrop     = ["internal_*"]
Filtered /tmp/telegraf-46-filt.conf
# Filter A + Filter B + pivot

[agent]
  interval          = "1s"
  precision         = "1ms"
  flush_interval    = "5s"
  omit_hostname     = true

[[inputs.mqtt_consumer]]
  servers    = ["tcp://localhost:1883"]
  topics     = ["ems/#"]
  client_id  = "telegraf-46-filt"
  data_format = "json_v2"

    # ── Filter A: drop string dtype at input ──────────
    [inputs.mqtt_consumer.tagdrop]
    data_type = ["string"]
    # ──────────────────────────────────────────────────

  [[inputs.mqtt_consumer.json_v2]]
    timestamp_path   = "ts"
    timestamp_format = "2006-01-02T15:04:05.000Z"

    [[inputs.mqtt_consumer.json_v2.field]]
      path = "value"

        # 9-seg topic parsing
    [[inputs.mqtt_consumer.topic_parsing]]
      topic = "ems/site/+/unit/+/+/+/+/+"
      tags  = "_/_/site_id/_/unit_id/device/instance/point_name/data_type"
      measurement = "_/_/_/measurement/_/_/_/_/_"

    # ── Filter B: dedup integer+boolean, order=0 ──────
  [[processors.dedup]]
    order          = 0
    dedup_interval = "60s"
    [processors.dedup.tagpass]
      data_type = ["integer", "boolean_integer"]
    # ──────────────────────────────────────────────────

[[processors.pivot]]
  order     = 1
  tag_key   = "data_type"
  value_key = "value"

[[outputs.influxdb_v2]]
  urls         = ["http://localhost:8097"]
  bucket       = "ems-filtered"
  namedrop     = ["internal_*"]
Filter A — drop data_type=string rows at input (~4% reduction)
Filter B — dedup integer/boolean_integer 60s window (~23% reduction)
